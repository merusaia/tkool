Sprite_Damage.prototype.setup = function(target) { // minRPG_CriticalEffectでは、クリティカル演出を変更可能にするため上書き。（デフォルトは、赤くフラッシュするだけ）
    var result = target.result();
    if (result.missed || result.evaded) {
        this.createMiss();
    } else if (result.hpAffected) {
        this.createDigits(0, result.hpDamage);
    } else if (target.isAlive() && result.mpDamage !== 0) {
        this.createDigits(2, result.mpDamage);
    }
    if (result.critical) {
        this.setupCriticalEffect();
    }
};

Sprite_Damage.prototype.setupCriticalEffect = function() { // minRPG_CriticalEffectでは、デフォルトのクリティカル演出は使ってないよ。上書きはしていないよ。
    this._flashColor = [255, 0, 0, 160];
    this._flashDuration = 60;
};
